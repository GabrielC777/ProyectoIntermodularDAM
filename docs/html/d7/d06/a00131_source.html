541 /// </summary>
542 private void CargarDatosEditarCancion()
543 {
544 ListaArtistasEditar.Clear();
545 ListaGenerosSeleccionadosEditar.Clear();
546 ListaResultadosEditar.Clear();
547 TxtBusquedaEditar = "";
548 HayResultadosEditar = false;
549
550 if (SelectedCancion != null)
551 {
552 // Listas
553 if (SelectedCancion.Datos != null && SelectedCancion.Datos.Generos != null)
554 foreach (var g in SelectedCancion.Datos.Generos) ListaGenerosSeleccionadosEditar.Add(g);
555
556 if (SelectedCancion.AutoresIds != null)
557 {
558 foreach (var id in SelectedCancion.AutoresIds)
559 {
560 var user = ListaUsuarios.FirstOrDefault(u => u.Id == id);
561 if (user != null) ListaArtistasEditar.Add(user);
562 }
563 }
564
565 // Audio (Detectar Youtube vs Local)
566 if (!string.IsNullOrEmpty(SelectedCancion.UrlCancion))
567 {
568 bool esLink = SelectedCancion.UrlCancion.StartsWith("http", StringComparison.OrdinalIgnoreCase) ||
569 SelectedCancion.UrlCancion.StartsWith("www", StringComparison.OrdinalIgnoreCase);
570
571 if (esLink)
572 {
573 EsArchivoLocalEditar = false; TxtUrlYoutubeEditar = SelectedCancion.UrlCancion; TxtRutaArchivoEditar = string.Empty;
574 }
575 else
576 {
577 EsArchivoLocalEditar = true; TxtRutaArchivoEditar = SelectedCancion.UrlCancion; TxtUrlYoutubeEditar = string.Empty;
578 }
579 }
