157 /// <summary>
158 /// Descarga de forma asíncrona una imagen desde una dirección URL y la asigna al mapa de bits de la portada.
159 /// </summary>
160 /// <remarks>
161 /// Este método gestiona la recuperación de recursos remotos mediante los siguientes pasos:
162 /// <list type="number">
163 /// <item><b>Petición HTTP:</b> Utiliza un <see cref="HttpClient"/> para obtener el flujo de bytes de la imagen desde la red.</item>
164 /// <item><b>Procesamiento de Memoria:</b> Transfiere los bytes a un <see cref="System.IO.MemoryStream"/> para su decodificación.</item>
165 /// <item><b>Asignación Visual:</b> Inicializa la propiedad <see cref="ImagenPortada"/> con el nuevo <see cref="Bitmap"/> y actualiza el estado de <see cref="TieneImagen"/>.</item>
166 /// </list>
167 /// En caso de error en la red o formato inválido, se captura la excepción y se notifica al usuario mediante <see cref="_dialogoService"/>.
168 /// </remarks>
169 /// <param name="url">La dirección URL completa de la imagen que se desea cargar.</param>
170 /// <returns>Una tarea que representa la operación de carga asíncrona.</returns>
171 private async Task CargarImagenDesdeUrl(string url)
172 {
173 if (string.IsNullOrEmpty(url)) return;
174 try
175 {
176 using (var client = new HttpClient())
177 {
178 var bytes = await client.GetByteArrayAsync(url);
179 using (var stream = new System.IO.MemoryStream(bytes))
180 {
181 ImagenPortada = new Bitmap(stream);
182 TieneImagen = true;
183 }
184 }
185 }
186 catch {
187 _dialogoService.MostrarAlerta("Msg_Error_CargarImagen");
188 }
189 }

283 /// <summary>

285 /// </summary>
286 /// <remarks>
287 /// Este método orquesta la actualización de la playlist mediante el siguiente flujo de trabajo:
288 /// <list type="number">
289 /// <item><b>Sincronización de Imagen:</b> Evalúa si la ruta de la portada es local o remota. En caso de ser local, sube el archivo a la nube mediante <see cref="_storageService"/> para obtener una URL persistente.</item>

292 /// <item><b>Finalización:</b> Tras el éxito, libera el estado de carga y ejecuta la acción de retorno a la vista anterior.</item>
293 /// </list>

297 private async Task GuardarCambios()
298 {
299 EstaCargando = true;

301 {
302 string urlPortadaFinal = RutaImagen;

304 //Si la imagen es local, la subimos y obtenemos la URL de la nube

306 {
307 urlPortadaFinal = await _storageService.SubirImagen(RutaImagen);
308 }
309
310 // Extraeremos IDs de canciones
311 var nuevosIds = ListaCancionesSeleccionadas.Select(c => c.Id).ToList();
312
313 // Actualizamos en la BD
314 await MongoClientSingleton.Instance.Cliente.ActualizarPlaylist(
315 TxtNombre,
316 TxtDescripcion,
317 nuevosIds,
318 urlPortadaFinal,
319 _playlistOriginal
320 );
321
322 EstaCargando = false;
323 _accionVolver();
324 }
325 catch (Exception ex)
326 {
327 EstaCargando = false;
328 _dialogoService.MostrarAlerta("Msg_Error_ActualizarPlaylist");
329 System.Diagnostics.Debug.WriteLine($"Error al actualizar playlist: {ex.Message}");
