98 /// <summary>
99 /// Dirige la navegación a la vista de un artista cuando se activa mediante un botón asociado con el nombre del artista.
100 /// </summary>
101 /// <remarks>Este método se utiliza típicamente como un manejador de comandos para elementos de la interfaz de usuario que representan
102 /// artistas dentro de un contexto de canción. Recupera el artista y la información de la canción relevante desde el botón
103 /// jerarquía y plantea una solicitud para mostrar los detalles del artista. El parámetro debe estructurarse como
104 /// descrito para que la navegación tenga éxito. </remarks>
105 /// <param name="parametro">El parametro de comando, que se espera sea un botón cuyo DataContext contiene el nombre del artista y cuya etiqueta
106 /// hace referencia al botón padre que contiene la información de la canción. </param>
107 private void IrAArtistaDesdeBoton(object parametro)
108 {
109 // Recuperamos el objeto de donde viene el comando (El botón con el nombre del artista)
110 // y lo casteamos a Button para sacarle la inforamación que necesitamos
111 if (parametro is Button botonPequeño)
112 {
113 // Recuperamos el Nombre del Artista (DataContext del botón (Que seria un Objeto Canciones))
114 var nombreArtista = botonPequeño.DataContext as string;
115
116 // Recuperamos el botón "jefe" (el abre el menú contextual donde esta el nombre del artistas)
117 // Que guardamos su referencia en el Tag del botón pequeño
118 if (botonPequeño.Tag is Button botonJefe)
119 {
120 //Ahora le decimos que se oculte par que no se quede abierto al pasar la vista artista
121 botonJefe.Flyout?.Hide();
122
123 // Y recuperamos la canción (DataContext del botón jefe)
124 var cancion = botonJefe.DataContext as Canciones;
125
126 // Aseguramos que no esten vacíos
127 if (!string.IsNullOrEmpty(nombreArtista) && cancion != null)
128 {
129 // Buscamos el índice del artista en la lista del artista que queremos mostrar
130 int indice = cancion.ListaArtistasIndividuales.IndexOf(nombreArtista);
131 // Con ese índice, buscamos el ID del artista en la lista de IDs (que es paralela a la de nombres)
132 if (indice >= 0 && cancion.AutoresIds != null && indice < cancion.AutoresIds.Count)
133 {
134 string idUsuario = cancion.AutoresIds[indice];//Sacamos el id del artista a mostrar
135 SolicitudVerArtista?.Invoke(idUsuario);// Enviamos la solicitud para mostrar la vista del artista con su ID
136 }
137 }
138 }
139 }

141 /// <summary>
142 /// Maneja el comando de reproducción disparado desde un botón, iniciando la reproducción de la canción seleccionada y su
143 /// colección asociada.
144 /// </summary>
145 /// <remarks>Este método se utiliza típicamente como un manejador de comandos para botones de reproducción en el usuario
146 /// interfaz. El DataContext del botón debe hacer referencia a la canción que se va a reproducir, y su etiqueta debe hacer referencia al
147 /// colección a la que pertenece la canción. Si falta alguno de los valores o es inválido, el método lo hace
148 /// nada. </remarks>
149 /// <param name="parametro">El parámetro de comando, que se espera sea un botón cuyo DataContext es una canción a reproducir y cuya etiqueta contiene
150 /// la colección de canciones. No debe ser nula y debe ser un botón con valores válidos de DataContext y Tag. </param>
151 private void ReproducirDesdeBoton(object parametro)
152 {
153 // Recuperamos el objeto de donde viene el comando (El botón de Play)
154 // y lo casteamos a Button para sacar la información que necesitamos
155 if (parametro is Button boton)
156 {
157 // Recuperamos la Canción a reproducir (DataContext del botón (Que sería el Objeto Canciones))
158 var cancion = boton.DataContext as Canciones;
159
160 // Recuperamos la lista completa a la que pertenece esa canción (la lista de origen)
161 // Que guardamos en el Tag del botón para saber el contexto (si viene de Populares, Buscador, etc.)
162 // Viene como 'IEnumerable', así que la casteamos para poder trabajar con ella
163 var coleccionOrigen = boton.Tag as IEnumerable<Canciones>;
164
165 // Aseguramos que hemos recuperado bien tanto la canción como su lista de origen
166 if (cancion != null && coleccionOrigen != null)
167 {
168 // Convertimos la colección a una Lista concreta para asegurarnos de pasar una copia exacta al reproductor
169 var listaExacta = coleccionOrigen.ToList();
170
171 // Enviamos la solicitud de reproducción al padre (MarcoApp) pasando:
172 // 1. La canción concreta que se ha pulsado (para que empiece por ahí)
173 // 2. La lista completa de contexto (para que sepa qué poner cuando acabe esta)
174 EnviarReproduccion?.Invoke(cancion, listaExacta);
175 }
176 }
177 }
178 /// <summary>
179 /// Inicia la reproducción de la lista de reproducción especificada, reproduciendo la primera canción y poniendo en fila las canciones restantes para

187 if (playlist != null && playlist.CancionesCompletas.Count > 0)

189 // La primera canción que sonará
190 var primeraCancion = playlist.CancionesCompletas[0];
191

193 var cola = playlist.CancionesCompletas;
194

196 EnviarReproduccion?.Invoke(primeraCancion, cola);

204 /// <summary>
205 /// Carga y organiza de forma asíncrona los datos de canciones y listas de reproducción desde la base de datos, actualizando los correspondientes
206 /// colecciones para la vinculación de datos.
207 /// </summary>
208 /// <remarks>Si la conexión a la base de datos no está disponible, se muestra una alerta y no hay datos
209 /// cargado. Las canciones y listas de reproducción se agrupan en secciones como favoritas, nuevos lanzamientos, rock, personalizadas
210 /// listas y listas de comunidades, asignadas a sus respectivas colecciones para la vinculación de la interfaz. </remarks>
211 /// <returns>Devuelve una tarea que representa la operación de carga asíncrona. </returns>
212 private async Task CargarDatosCanciones()
213 {
214 if (MongoClientSingleton.Instance.Cliente == null)
215 {
216 // "Error de conexión a la base de datos"
217 _dialogoService.MostrarAlerta("Msg_Error_Conexion");
218 }
219 else
220 {
221 var cliente = MongoClientSingleton.Instance.Cliente;
222 var miIdUsuario = GlobalData.Instance.UserIdGD;
223
224 // Lógica de carga
225 //-Listade canciones
226 var songsNovedades = cliente.ObtenerCacionesNovedades();
227 var songsFavoritos = cliente.ObtenerCancionesFavoritos();
228 var songsRock = cliente.ObtenerCancionesPorGenero("Rock");
229 var songsGeneral = cliente.ObtenerCanciones();
230 //-Playlist
231 var taskPlaylists = cliente.ObtenerListasReproduccion();
232
233 // Esperamos todo a la vez (Paralelismo)
234 await Task.WhenAll(songsNovedades, songsFavoritos, songsRock, songsGeneral, taskPlaylists);
235
236 // Creamos la lista de secciones
237 var listaTarjetas = new ObservableCollection<TarjetasCanciones>();
238
239 // --- PROCESAR CANCIONES (TarjetasCanciones) ---
240 // OJO: Estas claves "Sec_..." vienen de un Diccionario si posieramos el texto directamente,
241 // se podria direcamente esa palabra pero no cambiaria entre idiomas
242
243 if (songsFavoritos.Result.Count > 0)
244 listaTarjetas.Add(new TarjetasCanciones("Sec_Favoritos", new ObservableCollection<Canciones>(songsFavoritos.Result))); // "Favoritos "
245
246 listaTarjetas.Add(new TarjetasCanciones("Sec_Novedades", new ObservableCollection<Canciones>(songsNovedades.Result))); // "Novedades "
247
248 if (songsRock.Result.Count > 0)
249 listaTarjetas.Add(new TarjetasCanciones("Sec_Rock", new ObservableCollection<Canciones>(songsRock.Result))); // "Rock "
250
251 listaTarjetas.Add(new TarjetasCanciones("Sec_ParaTi", new ObservableCollection<Canciones>(songsGeneral.Result))); // "Para ti "
252
253
254 Tarjetas = listaTarjetas;// Asignamos la lista al Binding
255
256 // --- PROCESAR LISTAS (TarjetasListas) ---
257 // Aquí separamos las listas en secciones
258 var todasListas = taskPlaylists.Result;
259 var listaPlaylist = new ObservableCollection<TarjetasListas>();
260
261 // Mis Listas (Filtramos por mi ID)
262 var misListas = todasListas.Where(l => l.IdUsuario == miIdUsuario).ToList();
263 if (misListas.Count > 0)
264 {
265 listaPlaylist.Add(new TarjetasListas("Sec_MisListas", new ObservableCollection<ListaPersonalizada>(misListas))); // "Mis Listas 👤"
266 }
267
268 // Comunidad (El resto)
269 var otrasListas = todasListas.Where(l => l.IdUsuario != miIdUsuario).ToList();
270 if (otrasListas.Count > 0)
271 {
272 listaPlaylist.Add(new TarjetasListas("Sec_Comunidad", new ObservableCollection<ListaPersonalizada>(otrasListas))); // "De la Comunidad 🌎"
273 }
274
275 Playlists = listaPlaylist;// Asignamos la lista al Binding
276
277 }
278 }
