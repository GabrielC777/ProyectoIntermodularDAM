123 /// <summary>
124 /// Recupera todos los reportes de la base de datos y los clasifica en colecciones independientes según su estado actual.
125 /// </summary>
126 /// <remarks>
127 /// Este método realiza una limpieza integral de las listas y selecciones actuales para evitar duplicidad visual.
128 /// Posteriormente, consulta MongoDB y distribuye cada reporte en las categorías de "Pendiente", "Investigando"
129 /// o "Finalizado" basándose en el valor de su propiedad <c>Estado</c>, facilitando la organización por columnas en la interfaz.
130 /// </remarks>
131 /// <returns>Una tarea que representa la operación de carga y clasificación asíncrona.</returns>
132 private async Task CargarDatos()
133 {
134 // Limpiamos todo
135 ListaPendientes.Clear();
136 ListaInvestigando.Clear();
137 ListaFinalizados.Clear();
138
139 // Limpiamos selecciones
140 SelectedPendiente = null;
141 SelectedInvestigando = null;
142 SelectedFinalizado = null;
143 ReporteSeleccionado = null;
144
145 var todos = await MongoClientSingleton.Instance.Cliente.ObtenerReportes();
146
147 foreach (var r in todos)
148 {
149 switch (r.Estado?.Trim())
150 {
151 case "Pendiente": ListaPendientes.Add(r); break;
152 case "Investigando": ListaInvestigando.Add(r); break;
153 case "Finalizado": ListaFinalizados.Add(r); break;
154 default: ListaPendientes.Add(r); break;
155 }
156 }
157 }
158 /// <summary>
159 /// Persiste de forma asíncrona las modificaciones realizadas en el estado y la resolución del reporte seleccionado.
160 /// </summary>
161 /// <remarks>
162 /// Este método sincroniza los cambios con la base de datos MongoDB mediante los siguientes pasos:
163 /// <list type="number">
164 /// <item><b>Validación:</b> Verifica que exista una instancia válida en <see cref="ReporteSeleccionado"/>.</item>
165 /// <item><b>Sincronización:</b> Envía los nuevos valores de <c>EstadoEdit</c> y <c>ResolucionEdit</c> al servidor.</item>
166 /// <item><b>Refresco:</b> Si la operación es exitosa, notifica al usuario y reejecuta <see cref="CargarDatos"/> para reorganizar los reportes en sus respectivas columnas visuales.</item>
167 /// </list>
168 /// </remarks>
169 /// <returns>Una tarea que representa la operación de actualización asíncrona.</returns>
170 private async Task GuardarCambios()
171 {
172 if (ReporteSeleccionado == null) return;
173
174 bool exito = await MongoClientSingleton.Instance.Cliente.ActualizarEstadoReporte(
175 EstadoEdit, ResolucionEdit, ReporteSeleccionado
176 );
177
178 if (exito)
179 {
180 _dialogoService.MostrarAlerta("Reportes_MsgActualizado");
181 await CargarDatos();
182 }
183 else
184 {
185 _dialogoService.MostrarAlerta("Reportes_MsgSinCambios");
186 }
187 }
188 }
