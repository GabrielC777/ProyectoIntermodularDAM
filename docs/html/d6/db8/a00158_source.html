34 /// <summary>
35 /// Configura el entorno de ejecución local, asegurando la presencia de las dependencias binarias y sanitizando archivos de configuración.
36 /// </summary>
37 /// <remarks>
38 /// Este método de preparación realiza las siguientes operaciones críticas:
39 /// <list type="number">
40 /// <item><b>Despliegue de Binarios:</b> Verifica la existencia de <c>yt-dlp.exe</c> en la ruta de ejecución. Si no está presente o el archivo está dañado (0 bytes), realiza una copia desde el directorio de instalación.</item>
41 /// <item><b>Sanitización de Cookies:</b> Procesa el archivo <c>cookies.txt</c> para eliminar la marca de orden de bytes (BOM). Esto es indispensable ya que <c>yt-dlp</c> requiere una codificación UTF-8 pura para validar sesiones de usuario.</item>
42 /// <item><b>Normalización de Rutas:</b> Centraliza los archivos operativos en carpetas de datos de usuario para evitar problemas de permisos de escritura.</item>
43 /// </list>
44 /// Cualquier error durante el acceso a archivos o escritura de disco se captura y se registra en la consola para facilitar el diagnóstico.
45 /// </remarks>
46 /// <param name="appDir">Directorio raíz donde se encuentran los archivos originales de la aplicación.</param>

90 /// <summary>
91 /// Ejecuta el comando de auto-actualización del binario <c>yt-dlp</c> de forma silenciosa.
92 /// </summary>
93 /// <remarks>
94 /// Dado que las plataformas de video cambian sus algoritmos frecuentemente, este método asegura que la herramienta
95 /// de extracción esté en su versión más reciente mediante los siguientes pasos:
96 /// <list type="number">
97 /// <item><b>Validación:</b> Comprueba la existencia del ejecutable antes de intentar la actualización.</item>
98 /// <item><b>Ejecución en segundo plano:</b> Inicia un proceso externo con el argumento <c>--update</c> configurado para no mostrar ventanas (<c>CreateNoWindow</c>).</item>
99 /// <item><b>Redirección:</b> Captura las salidas del proceso para evitar bloqueos y espera su finalización síncrona.</item>
100 /// <item><b>Resiliencia:</b> El bloque <c>catch</c> ignora silenciosamente fallos (como falta de internet o bloqueos de firewall) para permitir que la aplicación principal siga funcionando incluso si la actualización falla.</item>
101 /// </list>
102 /// </remarks>
103 private void ActualizarYtDlp()
104 {
105 try
106 {
107 if (!System.IO.File.Exists(_ytDlpPath)) return;
108 Console.WriteLine("[API] Buscando actualizaciones de yt-dlp...");
109
110 var psi = new ProcessStartInfo
111 {
112 FileName = _ytDlpPath,
113 Arguments = "--update",
114 RedirectStandardOutput = true,
115 RedirectStandardError = true,
116 UseShellExecute = false,
117 CreateNoWindow = true
118 };
119
120 using var process = Process.Start(psi);
121 process.WaitForExit();
122 Console.WriteLine("[API] Actualización completada.");
123 }
124 catch { /* Ignorar errores de red */ }
125 }

128 /// <summary>
129 /// Procesa una solicitud HTTP GET para extraer la URL de streaming directo y los metadatos de un video de YouTube.
130 /// </summary>
131 /// <remarks>
132 /// El flujo de ejecución de este endpoint es el siguiente:
133 /// <list type="number">
134 /// <item><b>Validación:</b> Comprueba que la URL recibida no sea nula y que el binario <c>yt-dlp</c> esté disponible en el servidor.</item>
135 /// <item><b>Preparación de Argumentos:</b> Configura <c>yt-dlp</c> con los parámetros <c>--dump-json</c> (para obtener datos en lugar de descargar) y <c>--cookies</c> (usando la versión sanitizada para evitar bloqueos).</item>
136 /// <item><b>Ejecución de Proceso:</b> Inicia un proceso externo de forma asíncrona, capturando la salida estándar codificada en UTF-8 para evitar errores con caracteres especiales.</item>
137 /// <item><b>Análisis de Datos:</b> Parsea la salida JSON generada por la herramienta para extraer el enlace directo (<c>url</c>) y la duración exacta en segundos (<c>duration</c>).</item>
138 /// </list>
139 /// </remarks>
140 /// <param name="url">La dirección URL del video de YouTube proporcionada como parámetro de consulta (query string).</param>
141 /// <returns>
142 /// Un objeto JSON que contiene la URL de streaming directo y la duración;
143 /// o un código de error (400 o 500) con el detalle del fallo.
144 /// </returns>
145 [HttpGet("stream")]
146 public async Task<IActionResult> GetStreamUrl([FromQuery] string url)
147 {
148 // Validamos que la URL no llegue vacía o con espacios
149 if (string.IsNullOrWhiteSpace(url)) return BadRequest("Falta la URL");
150
151 try
152 {
153 // Verificamos físicamente si el ejecutable yt-dlp está en la carpeta del servidor
154 if (!System.IO.File.Exists(_ytDlpPath)) return StatusCode(500, "FATAL: yt-dlp no instalado.");
155
156 // Preparamos los comandos para el ejecutable:
157 // --dump-json: No descarga, solo devuelve datos técnicos.
158 // --no-playlist: Ignora listas, solo procesa el video del enlace.
159 // -f bestaudio: Busca el enlace con la mejor calidad de sonido disponible.
160 string argumentos = $"--dump-json --no-playlist -f bestaudio \"{url}\"";
161
162 // Si tenemos el archivo de cookies (ya saneado), lo añadimos para evitar bloqueos de YouTube
163 if (System.IO.File.Exists(_cookiesPath))
164 {
165 argumentos += $" --cookies \"{_cookiesPath}\"";
166 }
167
168 // Configuramos cómo se va a lanzar el proceso externo (yt-dlp.exe)
169 var psi = new ProcessStartInfo
170 {
171 FileName = _ytDlpPath, // Ruta del .exe
172 Arguments = argumentos, // Los comandos de arriba
173 RedirectStandardOutput = true, // Permite que la API lea lo que el programa escribe
174 RedirectStandardError = true, // Permite leer errores si los hay
175 UseShellExecute = false, // Obligatorio para redirigir flujos de datos
176 CreateNoWindow = true, // No abre la ventana negra de consola
177 StandardOutputEncoding = Encoding.UTF8 // Asegura que tildes y caracteres raros se lean bien
178 };
179 //Iniciamos el proceso
180 using var process = new Process { StartInfo = psi };
181 process.Start();
182
183 //Leemos de forma asíncrona toda la respuesta JSON que genera yt-dlp
184 string jsonOutput = await process.StandardOutput.ReadToEndAsync();
185
186 //Esperamos a que el programa termine de cerrarse
187 await process.WaitForExitAsync();
188
189 //Si el programa falló (ExitCode != 0) o no devolvió nada, avisamos del error
190 if (process.ExitCode != 0 || string.IsNullOrWhiteSpace(jsonOutput))
191 {
192 return StatusCode(500, "Error obteniendo audio. Revisa cookies o bloqueo regional.");
193 }
194
195 // Convertimos el texto JSON en un objeto manipulable en C#
196 var nodo = JsonNode.Parse(jsonOutput);
197
198 // Devolvemos un objeto limpio con solo los dos datos que le importan al cliente:
199 // La URL directa del flujo de audio y la duración en segundos.
200 return Ok(new
201 {
202 url = nodo["url"]?.ToString(),
203 duracion = nodo["duration"]?.GetValue<int>() ?? 0
204 });
205 }
206 catch (Exception ex)
207 {
208 Console.WriteLine($"[API CRITICAL] {ex.Message}");
209 return StatusCode(500, ex.Message);
210 }
211 }
212 }
