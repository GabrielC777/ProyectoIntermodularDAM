46 [HttpPost("subir-imagen")] //ruta --> api/Storage/subir-imagen
47 public async Task<IActionResult> SubirImagen(IFormFile archivo)
48 {
49 // Verificación de seguridad: ¿El archivo existe y tiene datos?
50 if (archivo == null || archivo.Length == 0) return BadRequest("No hay imagen");
51
52 try
53 {
54 // 'using': Crea un cliente HTTP (un navegador web invisible) y lo destruye al terminar para ahorrar RAM.
55 using (var client = new HttpClient())
56 {
57 // ImgBB necesita Base64
58 // Variable para guardar la imagen convertida a texto.
59 string base64Image;
60
61 // Abrimos un flujo de memoria temporal en la RAM.
62 using (var ms = new MemoryStream())
63 {
64 // Copiamos el archivo que llegó de internet(nuestra aplicación) a la memoria RAM.
65 await archivo.CopyToAsync(ms);
66 //Convertimos los bytes de la imagen a una cadena de texto Base64.
67 base64Image = Convert.ToBase64String(ms.ToArray());
68 }
69 // Preparamos el "formulario" virtual para enviar.
70 var content = new FormUrlEncodedContent(new[]
71 {
72 // Campo 1: La clave API.
73 new KeyValuePair<string, string>("key", ImgbbApiKey),
74 // Campo 2: La imagen convertida a texto.
75 new KeyValuePair<string, string>("image", base64Image)
76 });
77
78 // Enviamos la petición POST a la URL de ImgBB con el formulario.
79 var response = await client.PostAsync("https://api.imgbb.com/1/upload", content);
80 // Leemos la respuesta del servidor (que viene en texto JSON).
81 var jsonString = await response.Content.ReadAsStringAsync();
82
83 // Si ImgBB dice que algo salió mal (código 400 o 500)...
84 if (!response.IsSuccessStatusCode)
85 return StatusCode(500, "Error ImgBB: " + jsonString);
86
87 // Analizamos el texto JSON para convertirlo en objeto.
88 var json = JObject.Parse(jsonString);
89
90 // Buscamos dentro del JSON: objeto 'data' -> propiedad 'url'.
91 string urlFinal = json["data"]["url"].ToString();
92
93 // Devolvemos un código 200 OK con la URL lista para usar.
94 return Ok(new { url = urlFinal });
95 }
96 }
97 catch (Exception ex)
98 {
99 return StatusCode(500, "Error subiendo imagen: " + ex.Message);
100 }

196 /// <summary>
197 /// Analiza una URL de Cloudinary y extrae el identificador único (Public ID) necesario para operaciones de gestión.
198 /// </summary>
199 /// <remarks>
200 /// El método realiza un "parseo" quirúrgico de la URL siguiendo este algoritmo:
201 /// <list type="number">
202 /// <item><b>Localización:</b> Busca el segmento <c>/upload/</c> que separa la configuración del servidor de los datos del archivo.</item>
203 /// <item><b>Limpieza de Versión:</b> Omite el componente de versión (ej. <c>v17397... </c>) que Cloudinary genera automáticamente.</item>
204 /// <item><b>Extracción de Carpeta y Nombre:</b> Captura la ruta interna y el nombre del archivo.</item>
205 /// <item><b>Remoción de Extensión:</b> Elimina el sufijo del formato (ej. <c>.mp3</c>) para obtener el ID limpio que requiere la API de borrado.</item>
206 /// </list>
207 /// </remarks>
208 /// <param name="url">La dirección URL completa del recurso alojado en Cloudinary.</param>
209 /// <returns>El Public ID del recurso (incluyendo carpetas) o <c>null</c> si el formato de la URL es inválido.</returns>
210 private string ObtenerPublicId(string url)
211 {
212 try
213 {
214 var uri = new Uri(url);
215 string path = uri.AbsolutePath;
216 // Ejemplo: /dyyi9sb9v/video/upload/v1234/musicsearch_audios/cancion.mp3
217
218 int indexUpload = path.IndexOf("/upload/");
219 if (indexUpload == -1) return null;
220
221 // Cortamos todo hasta después de /upload/
222 string resto = path.Substring(indexUpload + 8);
223
224 // Saltamos la versión (v12345/...)
225 int indexSlash = resto.IndexOf('/');
226 string idConExt = resto.Substring(indexSlash + 1);
227
228 // Quitamos la extensión (.mp3 o .jpg)
229 int indexPunto = idConExt.LastIndexOf('.');
230 return idConExt.Substring(0, indexPunto);
231 }
232 catch { return null; }
233 }
