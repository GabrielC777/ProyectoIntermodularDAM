394 /// <summary>
395 /// Realiza el proceso completo de publicación de una canción, integrando subida de archivos, obtención de datos y persistencia.
396 /// </summary>
397 /// <remarks>
398 /// Este método orquesta un flujo complejo dividido en cuatro fases principales:
399 /// <list type="number">
400 /// <item><b>Subida de imagen:</b> Sube la portada seleccionada a la nube.</item>
401 /// <item><b>Gestión de audio:</b> Sube el archivo MP3 o procesa el enlace de YouTube para obtener la duración y URL final.</item>
402 /// <item><b>Creación de modelo:</b> Construye el objeto <see cref="Canciones"/> con autores y géneros seleccionados.</item>
403 /// <item><b>Persistencia:</b> Guarda la canción en la BD y actualiza el contador de canciones del usuario.</item>
404 /// </list>

408 private async Task PublicarCancion()

411
412 try
413 {

419

422 {
423 var infoYoutube = await _audioService.ObtenerMp3(LinkYoutube);
424
425 if (infoYoutube != null)

427 urlAudioFinal = LinkYoutube; // Guardamos el enlace original
428 duracionFinal = infoYoutube.DuracionSegundos;
429 }
430 else
431 {
432 throw new InvalidOperationException("Msg_Error_InfoYoutube");
433 }
434 }
435 else
436 {

438 duracionFinal = _duracionCalculada;

440

442 var nuevaCancion = new Canciones

444 Titulo = TxtTitulo,
445 AutoresIds = ListaArtistas.Select(u => u.Id).ToList(),

447 UrlCancion = urlAudioFinal,
448 Datos = new DatosCancion

451 DuracionSegundos = duracionFinal,
452 Generos = ListaGenerosSeleccionados.ToList() // Guardamos la lista completa

455
456 // Guardamos en BD

459 if (exito)

469 EstaCargando = false;

472 }

475 EstaCargando = false;
476 _dialogoService.MostrarAlerta("Msg_Error_OperacionInvalida");
