16 /// <summary>
17 /// Traduce dinámicamente una clave de recurso en su valor correspondiente definido en los diccionarios de la aplicación.
18 /// </summary>
19 /// <remarks>
20 /// Este convertidor facilita la internacionalización (i18n) en la capa de vista mediante los siguientes pasos:
21 /// <list type="number">
22 /// <item><b>Validación:</b> Verifica si el valor de entrada es una cadena de texto válida (la clave del recurso).</item>
23 /// <item><b>Búsqueda:</b> Consulta el diccionario de recursos activo de <see cref="Application.Current"/> intentando localizar la clave.</item>
24 /// <item><b>Resolución:</b> Si encuentra el recurso (ej. una traducción o una ruta de imagen), lo devuelve; de lo contrario, retorna la clave original como valor de respaldo (fallback).</item>
25 /// </list>
26 /// Es ideal para enlazar propiedades de texto en XAML que deben reaccionar a cambios de idioma en tiempo de ejecución.
27 /// </remarks>
28 /// <param name="value">La clave del recurso (string) que se desea localizar.</param>
29 /// <param name="targetType">El tipo de la propiedad de destino.</param>
30 /// <param name="parameter">Parámetro opcional (no utilizado).</param>
31 /// <param name="culture">Información de cultura (no utilizada, se prioriza el diccionario activo).</param>
32 /// <returns>El objeto localizado encontrado en los recursos o el texto original si no existe coincidencia.</returns>
33 public object? Convert(object? value, Type targetType, object? parameter, CultureInfo culture)
34 {
35 // Si el valor es nulo o no es string, no hacemos nada
36 if (value is not string claveRecurso)
37 return value;
38
39 // Intentamos buscar la clave en los recursos de la App
40 if (Application.Current != null &&
41 Application.Current.TryGetResource(claveRecurso, null, out var recursoEncontrado))
42 {
43 return recursoEncontrado;
44 }
45
46 // Si no se encuentra (o si es un texto normal), devolvemos el texto original
47 return claveRecurso;
48 }
49 //ConvertBack no lo necesitamos porque solo es para mostrar texto, no para editarlo, así que devolvemos UnsetValue para indicar que no se puede convertir hacia atrás.

52 return BindingNotification.UnsetValue;
53 }
