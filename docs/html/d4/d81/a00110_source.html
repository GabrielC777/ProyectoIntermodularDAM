89 /// <summary>

92 /// <remarks>
93 /// Este método recupera los valores actuales de tema, idioma y tipografía desde <see cref="GlobalData.Instance"/>.
94 /// Posteriormente, traduce estas cadenas de texto a los índices correspondientes que utilizan los selectores de la vista
95 /// y fuerza la notificación de cambio de propiedades mediante <c>RaisePropertyChanged</c> para que la UI
96 /// refleje el estado real de la configuración.

98 private void CargarEstadoInicial()
99 {
100 // Usamos las variables de GlobalData para enseñar el estado actual de la configuración al usuario
101 var tema = GlobalData.Instance.DiccionarioTemaGD ?? "ModoClaro";

103 var fuente = GlobalData.Instance.DiccionarioFuenteGD ?? "Lexend";
104
105 // Sincronizar Tema
106 _indiceTema = (tema == "ModoClaro");
107 this.RaisePropertyChanged(nameof(IndiceTema));
108 this.RaisePropertyChanged(nameof(IndiceTemaOscuro));

110 // Sincronizar Idioma

112 this.RaisePropertyChanged(nameof(IndiceIdioma));

114 // Sincronizar Fuente

117 "Lexend" => 0,

127 /// <summary>
128 /// Ejecuta el cambio de apariencia visual de la aplicación entre modo claro y modo oscuro.
129 /// </summary>
130 /// <remarks>
131 /// Este método gestiona el cambio de tema en tres niveles:
132 /// <list type="number">
133 /// <item><b>Visual:</b> Aplica el diccionario de recursos de forma instantánea mediante <see cref="ControladorDiccionarios"/>.</item>
134 /// <item><b>Persistencia:</b> Si el tema es distinto al actual, sincroniza la preferencia en la base de datos MongoDB.</item>
135 /// <item><b>Estado Global:</b> Actualiza la propiedad en <see cref="GlobalData.Instance"/> para mantener la consistencia en toda la sesión.</item>
136 /// </list>
137 /// </remarks>

139 private void AplicarCambioTema(bool esClaro)
140 {
141 string nuevoTema = esClaro ? "ModoClaro" : "ModoOscuro";
142
143 //Aplicamos el tema en la app
144 ControladorDiccionarios.AplicarTema(nuevoTema);

146 // Comprobamos si realmente cambió respecto a GlobalData
147 if (GlobalData.Instance.DiccionarioTemaGD != nuevoTema)

149 // Guardamos en Mongo

218 /// Este método implementa una lógica de actualización parcial. Construye un objeto de configuración

226 private void GuardarConfiguracionEnMongo(string? temaNuevo, string? idiomaNuevo, string? fuenteNuevo)
227 {
228 if (string.IsNullOrEmpty(GlobalData.Instance.UserIdGD)) return;

235 DiccionarioFuente = fuenteNuevo ?? GlobalData.Instance.DiccionarioFuenteGD
236 };
237
238 // Ejecutamos en segundo plano para actualizar la configuraciñon del usuario.
239 _ = MongoClientSingleton.Instance.Cliente.ActualizarConfiguracionUsuario(GlobalData.Instance.UserIdGD, config);
