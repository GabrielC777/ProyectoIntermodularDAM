21 /// <summary>
22 /// Aplica un algoritmo de hashing criptográfico SHA-256 a una cadena de texto para proteger información sensible.
23 /// </summary>
24 /// <remarks>
25 /// Este proceso de seguridad transforma la contraseña mediante los siguientes pasos:
26 /// <list type="number">
27 /// <item><b>Codificación:</b> Convierte la cadena original en una secuencia de bytes utilizando el estándar UTF-8.</item>
28 /// <item><b>Cifrado:</b> Utiliza una instancia de <see cref="SHA256"/> para calcular un resumen único (hash) de 256 bits.</item>
29 /// <item><b>Representación:</b> Transforma los bytes resultantes en una cadena hexadecimal de longitud fija (64 caracteres) mediante un <see cref="StringBuilder"/>.</item>
30 /// <item><b>Gestión de Memoria:</b> Emplea la sentencia <c>using</c> para garantizar la liberación inmediata de los recursos criptográficos en la memoria RAM.</item>
31 /// </list>
32 /// Nota: El hashing es una operación unidireccional; no es posible revertir el resultado para obtener la contraseña original.
33 /// </remarks>
34 /// <param name="password">La contraseña en texto plano que se desea anonimizar.</param>
35 /// <returns>Una cadena de texto en formato hexadecimal que representa el hash único de la contraseña.</returns>
36 public static string HashPassword(string password)
37 {
38 // Creamos un objeto Sha257
39 // Usamos 'using' para que se limpie de la memoria RAM automáticamente al terminar
40 using (SHA256 sha256 = SHA256.Create())
41 {
42 // Convertimos de string a array de byte
43 byte[] bytesPassword = Encoding.UTF8.GetBytes(password);
44
45 // Calculamos el hash
46 byte[] bytesDelHash = sha256.ComputeHash(bytesPassword);
47
48 // Preparamos el StringBuilder para el foreach
49 StringBuilder builder = new StringBuilder();
50
51 // Y convertimos los bytes del hash a hexadecimal
52 foreach (byte b in bytesDelHash)
53 {
54 builder.Append(b.ToString("x2"));
55 }
56 // Y devolvermos el stringbuilder como string para compararlo
57 return builder.ToString();
58 }

61 // Recibe los bytes "limpios" de la canción y devuelve una "papilla" ilegible.
62 /// <summary>
63 /// Realiza un cifrado simétrico AES sobre una secuencia de bytes para proteger el contenido de archivos multimedia.
64 /// </summary>
65 /// <remarks>
66 /// Este proceso de encriptación transforma los datos originales en un formato ilegible mediante los siguientes pasos técnicos:
67 /// <list type="number">
68 /// <item><b>Inicialización:</b> Se instancia el algoritmo <see cref="Aes"/> y se configuran las propiedades <c>Key</c> (clave secreta) e <c>IV</c> (vector de inicialización).</item>
69 /// <item><b>Canalización (Streaming):</b> Se utiliza un <see cref="CryptoStream"/> como intermediario para procesar los datos a través de un transformador de cifrado.</item>
70 /// <item><b>Escritura Segura:</b> Los bytes originales se escriben en el flujo de memoria, donde se aplican las operaciones matemáticas del estándar AES.</item>
71 /// <item><b>Finalización:</b> Se ejecuta <c>FlushFinalBlock</c> para procesar los bytes restantes y garantizar la integridad del bloque cifrado.</item>
72 /// </list>
73 /// El resultado es un array de bytes que solo puede ser recuperado mediante el método de desencriptación correspondiente utilizando la misma clave e IV.
74 /// </remarks>
75 /// <param name="byteSinEncrip">El array de bytes original (en texto plano o formato multimedia crudo) que se desea proteger.</param>
76 /// <returns>Un array de bytes cifrados mediante el estándar AES, listos para ser almacenados de forma segura en el almacenamiento persistente.</returns>
77 public static byte[] EncriptarBytes(byte[] byteSinEncrip)
78 {
79 // Creamos el algoritmo AES (Estándar de cifrado simétrico)
80 using (Aes aes = Aes.Create())
81 {
82 aes.Key = _claveAes; // Asignamos nuestra clave secreta
83 aes.IV = _ivAes; // Y el vector de inicio
84
85 // Preparamos un flujo en memoria para guardar el resultado
86 using (var memoryStream = new MemoryStream())
87 {
88 // CryptoStream es un "túnel" que cifra todo lo que pasa por él.
89 // CryptoStreamMode.Write: Lo usamos para ESCRIBIR datos y que salgan cifrados.
90 using (var cryptoStream = new CryptoStream(memoryStream, aes.CreateEncryptor(), CryptoStreamMode.Write))
91 {
92 // Escribimos los datos de la canción en el túnel
93 cryptoStream.Write(byteSinEncrip, 0, byteSinEncrip.Length);
94
95 // "FlushFinalBlock" asegura que el último trocito de datos se escriba bien
96 cryptoStream.FlushFinalBlock();
97
98 // Devolvemos el array de bytes cifrados (ilegibles)
99 return memoryStream.ToArray();
100 }
101 }
102 }

107 /// <summary>
108 /// Lee un archivo cifrado del almacenamiento local, lo descifra utilizando el algoritmo AES y guarda el resultado en una ubicación temporal.
109 /// </summary>
110 /// <remarks>
111 /// Este método es el pilar de la recuperación de datos protegidos y opera bajo los siguientes pasos:
112 /// <list type="number">
113 /// <item><b>Carga de Datos:</b> Recupera los bytes cifrados del disco mediante <see cref="File.ReadAllBytesAsync"/>.</item>
114 /// <item><b>Configuración Criptográfica:</b> Reinstancia el motor <see cref="Aes"/> asegurando el uso de la misma clave y vector de inicialización (IV) empleados durante la encriptación.</item>
115 /// <item><b>Procesamiento de Flujo:</b> Utiliza un <see cref="CryptoStream"/> en modo lectura (<see cref="CryptoStreamMode.Read"/>) que actúa como un filtro de transformación, convirtiendo los bytes cifrados en datos originales.</item>
116 /// <item><b>Persistencia Temporal:</b> Vuelca el flujo descifrado en un nuevo archivo físico (normalmente un .mp3 temporal) para que sea accesible por los servicios de reproducción.</item>
117 /// </list>
118 /// Al utilizar flujos asíncronos, se garantiza que la interfaz de usuario no se bloquee durante el procesamiento de archivos de gran tamaño.
119 /// </remarks>
120 /// <param name="rutaEntrada">La ruta del archivo cifrado (habitualmente con extensión .enc) que se desea procesar.</param>
121 /// <param name="rutaSalida">La ruta de destino donde se escribirá el archivo resultante ya descifrado.</param>
122 /// <returns>Una tarea asíncrona que representa el proceso de lectura, descifrado y escritura.</returns>
123 public static async Task DesencriptarArchivo(string rutaEntrada, string rutaSalida)
124 {
125 // Volvemos a crear AES con la MISMA CLAVE (imprescindible en simétrico)
126 using (Aes aes = Aes.Create())
127 {
128 aes.Key = _claveAes;
129 aes.IV = _ivAes;
130
131 // 1. Leemos el archivo cifrado del disco (son bytes "basura" para un humano)
132 byte[] bytesCifrados = await File.ReadAllBytesAsync(rutaEntrada);
133
134 // 2. Preparamos los flujos (Streams) para procesar los datos
135 using (var memoryStreamEntrada = new MemoryStream(bytesCifrados))
136 {
137 using (var memoryStreamSalida = new MemoryStream())
138 {
139 // 3. Creamos el túnel de descifrado.
140 // CryptoStreamMode.Read: Al LEER del túnel, los datos salen limpios.
141 using (var cryptoStream = new CryptoStream(memoryStreamEntrada, aes.CreateDecryptor(), CryptoStreamMode.Read))
142 {
143 // Copiamos los datos a través del túnel hacia la salida
144 await cryptoStream.CopyToAsync(memoryStreamSalida);
145 }
146
147 // 4. Guardamos el resultado "limpio" en el archivo de salida (temp)
148 await File.WriteAllBytesAsync(rutaSalida, memoryStreamSalida.ToArray());
149 }
150 }
151 }
152 }
