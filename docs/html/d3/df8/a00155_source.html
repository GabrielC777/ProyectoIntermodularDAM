129 /// <summary>
130 /// Cambia el estado de seguimiento del usuario cargado actualmente para el usuario activo. Si el usuario activo ya está
131 /// siguiendo al usuario cargado, este método dejará de seguir; de lo contrario, iniciará un seguimiento.
132 /// </summary>
133 /// <remarks>Si el usuario activo intenta seguirse a sí mismo, se muestra una alerta y no hay acción
134 /// se toma. El método actualiza tanto el estado de seguimiento como la lista local de seguidores al éxito. </remarks>
135 /// <returns>Devuelve una tarea que representa la operación asíncrona. </returns>
136 private async Task AlterarSeguimiento()
137 {
138 string miId = GlobalData.Instance.UserIdGD;
139
140 if (_idUsuarioCargado == miId)
141 {
142 // "No puedes seguirte a ti mismo"
143 _dialogoService.MostrarAlerta("Msg_Error_SeguirseMismo");
144 return;
145 }
146
147 bool exito;
148
149 if (EsSeguido)
150 {
151 exito = await MongoClientSingleton.Instance.Cliente.DejarDeSeguirUsuario(miId, _idUsuarioCargado);
152 if (exito)
153 {
154 EsSeguido = false;
155 // Actualizamos la lista local
156 Usuario.Listas.Seguidores?.Remove(_idUsuarioCargado);
157 }
158 }
159 else
160 {
161 exito = await MongoClientSingleton.Instance.Cliente.SeguirUsuario(miId, _idUsuarioCargado);
162 if (exito)
163 {
164 EsSeguido = true;
165 // Actualizamos la lista local
166 Usuario.Listas.Seguidores?.Add(_idUsuarioCargado);
167 }
168 }
169 }

190 /// <summary>
191 /// Inicia un ciclo de actualización en segundo plano que actualiza periódicamente el usuario y las listas relacionadas hasta que se cancele
192 /// solicitado.
193 /// </summary>
194 /// <remarks>El bucle de actualización se ejecuta de forma asíncrona y actualiza los elementos de la interfaz de usuario para reflejar el estado actual.
195 /// estado de actualización. El método no bloquea el hilo de llamada. Para detener el proceso de actualización, indica
196 /// cancelación a través del token proporcionado. </remarks>
197 /// <param name="token">Un token de cancelación que se puede usar para solicitar la finalización del ciclo de actualización. </param>
198 private void IniciarHiloActualizacion(CancellationToken token)
199 {
200 Task.Run(async () =>
201 {
202 try
203 {
204 //Carga inicial
205 await CargarUsuario();
206
207 // Si encontramos al usuario, cargamos sus listas
208 if (_usuario != null)
209 {
210 await CargarListasDetalladas(_usuario.Id);
211
212 }
213
214 //Bucle de actualización cada 5 segundos
215 while (!token.IsCancellationRequested)
216 {
217 for (int i = 5; i > 0; i--)
218 {
219 await Dispatcher.UIThread.InvokeAsync(() =>
220 {
221 TxtMensajeTimer = "VisorUser_Timer_Refrescando";
222 TxtVariableTimer = $" {i}s";
223 });
224 await Task.Delay(1000, token);
225 }
226
227 await Dispatcher.UIThread.InvokeAsync(() =>
228 {
229 TxtMensajeTimer = "VisorUser_Timer_Consultando";
230 TxtVariableTimer = "";
231 });
232
233 // Refrescamos usuario y listas
234 await CargarUsuario();
235 if (_usuario != null)
236 {
237 // Actualizamos la UI visualmente
238 await Dispatcher.UIThread.InvokeAsync(() =>
239 {
240 TxtMensajeTimer = "VisorUser_Timer_Actualizado";
241 TxtVariableTimer = "";
242 });
243 }
244 }
245 }
246 catch (TaskCanceledException) { }
247 });
248 }
