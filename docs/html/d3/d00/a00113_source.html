132 /// <summary>
133 /// Realiza una búsqueda asíncrona de canciones en la base de datos utilizando el texto introducido por el usuario.
134 /// </summary>
135 /// <remarks>
136 /// Este método consulta MongoDB y filtra los resultados obtenidos para excluir aquellas canciones
137 /// que ya están presentes en la lista de selección (<see cref="ListaCancionesSeleccionadas"/>).
138 /// Esto evita duplicados visuales y actualiza la colección de resultados disponibles para añadir.
139 /// </remarks>
140 private async void BuscarCanciones()
141 {
142 if (MongoClientSingleton.Instance.Cliente != null)
143 {
144 var resultados = await MongoClientSingleton.Instance.Cliente.ObtenerCancionesPorBusqueda(TxtBusqueda);
145
146 if (resultados != null)
147 {
148 // Filtramos las que ya están seleccionadas para no duplicar visualmente
149 var filtradas = resultados.Where(c => !ListaCancionesSeleccionadas.Any(sel => sel.Id == c.Id));
150 ListaResultados = new ObservableCollection<Canciones>(filtradas);
151 }
152 }
153 }

155 /// <summary>
156 /// Agrega la canción seleccionada a la lista temporal de canciones que formarán parte de la nueva playlist.
157 /// </summary>
158 /// <remarks>
159 /// Este método realiza tres acciones clave:
160 /// <list type="number">
161 /// <item>Verifica que la canción no esté ya añadida para evitar duplicados.</item>
162 /// <item>Mueve visualmente la canción: la añade a <see cref="ListaCancionesSeleccionadas"/> y la elimina de <see cref="ListaResultados"/>.</item>
163 /// <item>Limpia el campo de búsqueda para facilitar una nueva consulta inmediata.</item>
164 /// </list>
165 /// </remarks>
166 /// <param name="cancion">El objeto <see cref="Canciones"/> que el usuario ha seleccionado para añadir.</param>
167 private void AgregarCancion(Canciones cancion)
168 {
169 if (!ListaCancionesSeleccionadas.Any(c => c.Id == cancion.Id))
170 {
171 ListaCancionesSeleccionadas.Add(cancion);
172 ListaResultados.Remove(cancion); // La quitamos de resultados para que quede limpio
173 TxtBusqueda = "";
174 }
175 }

220 /// <summary>
221 /// Orquesta el proceso completo de creación de una nueva lista de reproducción personalizada de forma asíncrona.
222 /// </summary>
223 /// <remarks>
224 /// Este método sigue un flujo de transacciones paso a paso:
225 /// <list type="number">
226 /// <item><b>Carga de medios:</b> Sube la imagen de portada seleccionada al servicio de almacenamiento en la nube.</item>
227 /// <item><b>Construcción del modelo:</b> Crea una instancia de <see cref="ListaPersonalizada"/> con los metadatos y la selección de canciones actual.</item>
228 /// <item><b>Persistencia:</b> Invoca al cliente de MongoDB para guardar la nueva lista en la base de datos.</item>
229 /// </list>
230 /// Gestiona los estados de carga (<see cref="EstaCargando"/>) para bloquear la UI durante el proceso y maneja excepciones globales.
231 /// </remarks>
232 /// <returns>Una <see cref="Task"/> que representa la operación asíncrona.</returns>
233 private async Task CrearLista()
234 {
235 EstaCargando = true;
236 try
237 {
238 // Subir Imagen
239 string urlPortada = await _storageService.SubirImagen(RutaImagen);
240
241 // Crear Objeto
242 var nuevaLista = new ListaPersonalizada
243 {
244 Nombre = TxtNombre,
245 Descripcion = TxtDescripcion,
246 UrlPortada = urlPortada,
247 IdUsuario = GlobalData.Instance.UserIdGD,
248 IdsCanciones = ListaCancionesSeleccionadas.Select(c => c.Id).ToList()
249 };
250
251 // Guardarmos en BD
252 bool exito = await MongoClientSingleton.Instance.Cliente.CrearListaReproduccion(nuevaLista);
253
254 if (exito)
255 {
256 EstaCargando = false;
257 _Volver();
258 }
259 else
260 {
261 EstaCargando = false;
262 _dialogoService.MostrarAlerta("Msg_Error_CrearPlaylist");
263 }
264 }
265 catch (Exception ex)
266 {
267 EstaCargando = false;
268 _dialogoService.MostrarAlerta("Msg_Error_Inesperado");
269 System.Diagnostics.Debug.WriteLine("Error crear lista: " + ex.Message);
270 }
271 }
