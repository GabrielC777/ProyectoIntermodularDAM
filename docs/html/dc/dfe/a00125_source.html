218 /// <summary>
219 /// Descarga de forma asíncrona una imagen desde una dirección URL y la asigna al mapa de bits de la portada.
220 /// </summary>
221 /// <remarks>
222 /// Este método gestiona la recuperación de recursos remotos mediante los siguientes pasos:
223 /// <list type="number">
224 /// <item><b>Petición HTTP:</b> Utiliza un <see cref="HttpClient"/> para obtener el flujo de bytes de la imagen desde la red.</item>
225 /// <item><b>Procesamiento de Memoria:</b> Transfiere los bytes a un <see cref="System.IO.MemoryStream"/> para su decodificación.</item>
226 /// <item><b>Asignación Visual:</b> Inicializa la propiedad <see cref="ImagenPortada"/> con el nuevo <see cref="Bitmap"/> y actualiza el estado de <see cref="TieneImagen"/>.</item>
227 /// </list>
228 /// El método incluye un bloque try-catch silencioso para asegurar que fallos en la red o URLs inválidas no interrumpan la ejecución de la aplicación.
229 /// </remarks>
230 /// <param name="url">La dirección URL completa de la imagen que se desea cargar.</param>
231 /// <returns>Una tarea que representa la operación de carga asíncrona.</returns>
232 private async Task CargarImagenDesdeUrl(string url)
233 {
234 if (string.IsNullOrEmpty(url)) return;
235 try
236 {
237 using (var client = new HttpClient())
238 {
239 var bytes = await client.GetByteArrayAsync(url);
240 using (var stream = new System.IO.MemoryStream(bytes))
241 {
242 ImagenPortada = new Bitmap(stream);
243 TieneImagen = true;
244 }
245 }
246 }
247 catch {
248 _dialogoService.MostrarAlerta("Msg_Error_CargarImagen");
249 }
250 }

400 /// <summary>
401 /// Procesa y persiste de forma asíncrona las modificaciones realizadas en una canción existente, gestionando la actualización de medios y metadatos.
402 /// </summary>
403 /// <remarks>
404 /// Este método orquesta la actualización de la canción siguiendo un flujo transaccional:
405 /// <list type="number">
406 /// <item><b>Gestión de Imagen:</b> Detecta si la ruta de la imagen es local o remota. Si es local, procede a subir el nuevo archivo mediante <see cref="_storageService"/>.</item>
407 /// <item><b>Sincronización remota:</b> Envía los nuevos títulos, IDs de colaboradores y géneros al cliente de MongoDB para actualizar el registro físico.</item>
408 /// <item><b>Actualización de estado local:</b> Si la persistencia es exitosa, sincroniza los cambios en el objeto <c>_cancionOriginal</c> para asegurar la consistencia visual al regresar a la vista anterior.</item>
409 /// </list>
410 /// Durante el proceso, controla la propiedad <see cref="EstaCargando"/> para feedback visual y gestiona posibles excepciones mediante el servicio de diálogos.
411 /// </remarks>
412 /// <returns>Una tarea que representa la operación de guardado asíncrona.</returns>
413 private async Task GuardarCambios()
414 {
415 EstaCargando = true;
416 try
417 {
418 string urlPortadaFinal = RutaImagen;
419
420 // Si la ruta NO es web (es decir, es un archivo local C:\Users\...), subimos la nueva.
421 if (!RutaImagen.StartsWith("http"))
422 {
423 urlPortadaFinal = await _storageService.SubirImagen(RutaImagen);
424 }
425
426 // Recogemos los datos nuevos
427 var nuevosAutoresIds = ListaArtistas.Select(u => u.Id).ToList();
428 var nuevosGeneros = ListaGenerosSeleccionados.ToList();
429
430 // Y los actualizamos a mongo
431 bool exito = await MongoClientSingleton.Instance.Cliente.ActualizarCancion(
432 TxtTitulo,
433 urlPortadaFinal,
434 nuevosAutoresIds,
435 nuevosGeneros,
436 _cancionOriginal
437 );
438
439 if (exito)
440 {
441 // Actualizamos el objeto original para que los cambios se reflejen al volver a la vista anterior
442 _cancionOriginal.Titulo = TxtTitulo;
443 _cancionOriginal.ImagenPortadaUrl = urlPortadaFinal;
444 _cancionOriginal.AutoresIds = nuevosAutoresIds;
445
446 if (_cancionOriginal.Datos == null) _cancionOriginal.Datos = new DatosCancion();
447 _cancionOriginal.Datos.Generos = nuevosGeneros;
448
449 // Actualizar nombre artista principal (display)
450 if (ListaArtistas.Count > 0)
451 _cancionOriginal.NombreArtista = ListaArtistas[0].Username;
452
453 _accionVolver();
454 }
455 else
456 {
457 _dialogoService.MostrarAlerta("Msg_Error_GuardarCambios");
458 }
459 }
460 catch (Exception ex)
461 {
462 _dialogoService.MostrarAlerta("Msg_Error_Inesperado");
463 System.Diagnostics.Debug.WriteLine(ex.Message);
464 }
465 finally
466 {
467 EstaCargando = false;
468 }
469 }
