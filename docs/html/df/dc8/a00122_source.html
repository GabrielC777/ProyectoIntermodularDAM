98 /// <summary>
99 /// Procesa y persiste de forma asíncrona las modificaciones realizadas en el perfil del usuario tanto en la base de datos como en el estado global.
100 /// </summary>
101 /// <remarks>
102 /// Este método realiza una validación y transformación de datos antes de la persistencia:
103 /// <list type="number">
104 /// <item><b>Conversión:</b> Transforma el objeto <see cref="DateTimeOffset"/> de la interfaz a <see cref="DateTime"/> y el índice de privacidad a un valor booleano.</item>
105 /// <item><b>Sincronización remota:</b> Invoca al cliente de MongoDB para actualizar los documentos en la nube.</item>
106 /// <item><b>Actualización local:</b> Si la operación remota es exitosa, sincroniza los nuevos valores en <see cref="GlobalData.Instance"/> para mantener la consistencia en la sesión actual.</item>
107 /// </list>
108 /// Notifica el resultado de la operación al usuario mediante el servicio de diálogos y registra errores críticos en la consola de depuración.
109 /// </remarks>
110 /// <returns>Una tarea que representa la operación de guardado asíncrona.</returns>
111 private async Task GuardarCambios()
112 {
113 try
114 {
115 // Convertimos los datos de la vista a los formatos de BD
116 var fechaParaGuardar = FechaNacimiento?.DateTime ?? DateTime.Now;
117 bool esCuentaPrivada = (IndexPrivacidad == 0);

122 Email,
123 Pais,
124 fechaParaGuardar,
125 esCuentaPrivada
126 );
127

133 GlobalData.Instance.PaisGD = Pais;
134 GlobalData.Instance.FechaNacimientoGD = fechaParaGuardar;

137 _dialogoService.MostrarAlerta("MsgExitoActualizarPerfil");
