57 /// <summary>
58 /// Solicita de forma asíncrona la eliminación de un recurso almacenado en la nube a través de la API de almacenamiento.
59 /// </summary>
60 /// <remarks>
61 /// Este método gestiona la baja de archivos (imágenes o audio) siguiendo este flujo:
62 /// <list type="number">
63 /// <item><b>Validación:</b> Verifica que la URL proporcionada no sea nula o vacía.</item>
64 /// <item><b>Seguridad:</b> Configura un <see cref="HttpClientHandler"/> para omitir la validación de certificados SSL, permitiendo peticiones a entornos <c>localhost</c> con certificados autofirmados.</item>
65 /// <item><b>Petición:</b> Ejecuta un verbo HTTP <c>DELETE</c> enviando la URL del archivo como parámetro de consulta.</item>
66 /// <item><b>Confirmación:</b> Evalúa la respuesta de la API para confirmar si el archivo fue removido con éxito del proveedor externo (Cloudinary).</item>
67 /// </list>
68 /// </remarks>
69 /// <param name="urlCompleta">La dirección URL absoluta del recurso que se desea eliminar del almacenamiento remoto.</param>
70 /// <returns>
71 /// Una tarea que devuelve <c>true</c> si el servidor confirma la eliminación (Success Status Code);
72 /// de lo contrario, <c>false</c> si el recurso no existe o hubo un fallo en la comunicación.
73 /// </returns>
74 public async Task<bool> EliminarArchivo(string urlCompleta)
75 {
76 // Validación básica: Si no hay URL, no hay nada que borrar.
77 if (string.IsNullOrEmpty(urlCompleta)) return false;
78
79 try
80 {
81 // Configuración para aceptar certificados locales
82 var handler = new HttpClientHandler();
83 handler.ServerCertificateCustomValidationCallback = (message, cert, chain, errors) => true;
84
85 using (var client = new HttpClient(handler))
86 {
87 // LA LLAMADA (DELETE)
88 // Construimos la URL así: https://localhost:7500/api/Storage/eliminar?url=https://
89 var response = await client.DeleteAsync($"{ApiUrl}/eliminar?url={urlCompleta}");
90
91 // RESULTADO
92 // Devuelve true si la API respondió 200 OK (Borrado exitoso)
93 // Devuelve false si falló (400, 500, etc.)
94 return response.IsSuccessStatusCode;
95 }
96 }
97 catch (Exception ex)
98 {
99 // Si hay error de conexión, solo lo registramos y devolvemos false
100 System.Diagnostics.Debug.WriteLine("[StorageService] Error al eliminar: " + ex.Message);

106 /// <summary>
107 /// Realiza la carga física de un archivo hacia la API de almacenamiento mediante una petición POST de tipo multipart/form-data.
108 /// </summary>
109 /// <remarks>
110 /// Este método núcleo (core) orquestra la transferencia de archivos multimedia siguiendo este flujo:
111 /// <list type="number">
112 /// <item><b>Verificación local:</b> Valida la existencia del archivo en el sistema de archivos del cliente antes de iniciar la conexión.</item>
113 /// <item><b>Configuración de Seguridad:</b> Implementa un bypass de validación SSL para permitir el desarrollo en entornos locales con certificados no firmados.</item>
114 /// <item><b>Empaquetado (Multipart):</b> Abre el archivo como un flujo de datos (<see cref="StreamContent"/>) y lo encapsula en un contenedor compatible con formularios web.</item>
115 /// <item><b>Transmisión y Respuesta:</b> Ejecuta la petición asíncrona hacia el <paramref name="endpoint"/> especificado y procesa la respuesta JSON para extraer la URL persistente generada por el servidor.</item>
116 /// </list>
117 /// </remarks>
118 /// <param name="ruta">La ruta absoluta del archivo en el disco duro local.</param>
119 /// <param name="endpoint">El segmento final de la URL de la API (ej. "subir-imagen" o "subir-audio").</param>
120 /// <returns>La URL pública del archivo cargado en el servidor de almacenamiento.</returns>
121 /// <exception cref="FileNotFoundException">Se lanza si la ruta especificada no apunta a un archivo válido.</exception>
122 /// <exception cref="Exception">Encapsula errores de conectividad, rechazos de la API (códigos 4xx o 5xx) o fallos en el análisis del JSON.</exception>
123 private async Task<string> EnviarA_Api(string ruta, string endpoint)
124 {
125 // Comprobamos si existe el archivo
126 if (!File.Exists(ruta)) throw new FileNotFoundException("¡No encuentro el archivo en tu PC!");
127
128 try
129 {
130 // Como estamos en desarrollo, el certificado de seguridad de 'localhost' no es oficial.
131 // Esta línea le dice al código: "Confía en el servidor".
132 var handler = new HttpClientHandler();
133 handler.ServerCertificateCustomValidationCallback = (message, cert, chain, errors) => true;
134
135 // Creamos el cliente
136 using (var client = new HttpClient(handler))
137 {
138 // Preparamos el paquete
139 using (var content = new MultipartFormDataContent())
140 {
141 // Abrimos el archivo de tu disco duro para leerlo.
142 var fileStream = File.OpenRead(ruta);
143
144 // Convertimos el archivo en un flujo de datos (StreamContent) para enviarlo.
145 var streamContent = new StreamContent(fileStream);
146
147 // Le ponemos una etiqueta para decirle que un archivo de datos
148 streamContent.Headers.ContentType = MediaTypeHeaderValue.Parse("multipart/form-data");
149
150 // Metemos el archivo en el sobre.
151 // "archivo" -> Es el nombre EXACTO que espera nuestra API
152 content.Add(streamContent, "archivo", Path.GetFileName(ruta));
153
154 // El envío (POST)
155 // Este sale hacia: https://localhost:7500/api/Storage/subir-imagen (o audio)
156 var response = await client.PostAsync($"{ApiUrl}/{endpoint}", content);
157
158 //Comprueba si fue bien. Si falla, se lanza HttpRequestException.
159 response.EnsureSuccessStatusCode();
160
161 // Si llegamos aquí, es que todo fue bien (Código 200)
162 var jsonString = await response.Content.ReadAsStringAsync();
163 var json = JObject.Parse(jsonString);
164 return json["url"].ToString();
165 }
166 }
167 }
168 catch (HttpRequestException httpEx)
169 {
170 // Aquí atrapamos errores específicos de la web (404, 500...)
171 throw new Exception($"La API rechazó el archivo: {httpEx.Message}");
172 }
173 catch (Exception ex)
174 {
175 // Si explota la conexión (servidor apagado, sin internet, etc.)
176 throw new Exception($"Error conectando con la API: {ex.Message}");
177 }
178 }
