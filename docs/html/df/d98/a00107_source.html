95 /// <summary>
96 /// Inicia un bucle de actualización en segundo plano que actualiza periódicamente la información actual de la canción hasta su cancelación
97 /// es solicitado.
98 /// </summary>
99 /// <remarks>Este método ejecuta la lógica de actualización en un hilo en segundo plano y actualiza los elementos de la interfaz de usuario.
100 /// usando el despachador. El ciclo de actualización continúa hasta que se indica el token de cancelación proporcionado. Intención
101 /// para uso interno para mantener la interfaz de usuario sincronizada con los datos más recientes de las canciones. </remarks>
102 /// <param name="token">Un token de cancelación que se puede usar para solicitar la finalización del ciclo de actualización. </param>
103 private void IniciarHiloActualizacion(CancellationToken token)
104 {
105 Task.Run(async () =>
106 {
107 try
108 {
109 while (!token.IsCancellationRequested)
110 {
111 // --- CUENTA ATRÁS VISUAL ---
112 // En lugar de esperar 3000ms de golpe, hacemos 3 pasos de 1000ms
113 for (int i = 5; i > 0; i--)
114 {
115 // Actualizamos el texto en la UI
116 await Dispatcher.UIThread.InvokeAsync(() =>
117 {
118 // Aquí separamos: Clave + Valor
119 TxtMensajeTimer = "VisorCan_Timer_Actualizando";
120 TxtVariableTimer = $" {i} s";
121 });
122
123 // Esperamos 1 segundo
124 await Task.Delay(1000, token);
125 }
126
127 // Avisamos que estamos consultando
128 await Dispatcher.UIThread.InvokeAsync(() =>
129 {
130 TxtMensajeTimer = "VisorCan_Timer_Consultando";
131 TxtVariableTimer = ""; // Limpiamos el número
132 });
133
134 // --- CONSULTA A LA BASE DE DATOS ---
135 var listaUnaCancion = new List<string> { _cancion.Id };
136 var resultado = await MongoClientSingleton.Instance.Cliente.ObtenerCancionesPorListaIds(listaUnaCancion);
137
138 var cancionActualizada = resultado.FirstOrDefault();
139
140 if (cancionActualizada != null)
141 {
142 await Dispatcher.UIThread.InvokeAsync(() =>
143 {
144 // Actualizamos métricas
145 Cancion = cancionActualizada;
146 // Mensaje de éxito breve
147 TxtMensajeTimer = "VisorCan_Timer_Exito";
148 TxtVariableTimer = "";
149 });
150 }
151 }
152 }
153 catch (TaskCanceledException) { /* Hilo detenido */ }
154 });
155 }
