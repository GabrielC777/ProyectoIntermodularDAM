100 /// <summary>

108 /// <returns>Devuelve una tarea que representa la operación de registro asíncrono. </returns>
109 private async Task RegistrarseTask()
110 {
111 if (EstaCargando) return;
112 EstaCargando = true;
113
114 try
115 {
116 // Validaciones básicas
117 if (string.IsNullOrWhiteSpace(NuevoUsuario.Username) ||
118 string.IsNullOrWhiteSpace(NuevoUsuario.Email) ||
119 string.IsNullOrWhiteSpace(NuevoUsuario.Password) ||
120 string.IsNullOrWhiteSpace(ConfirmarPass) ||
121 string.IsNullOrWhiteSpace(NuevoUsuario.Perfil.Pais) ||
122 string.IsNullOrWhiteSpace(NuevoUsuario.Perfil.ImagenUrl)
123 )
124 {
125 _dialogoService.MostrarAlerta("Reg_Error_FaltanCampos");
126 EstaCargando = false;
127 return;
128 }
129
130 // Validar contraseñas coinicidentes
131 if (NuevoUsuario.Password != ConfirmarPass)
132 {
133 _dialogoService.MostrarAlerta("Reg_Error_PassNoCoinciden");
134 EstaCargando = false;
135 return;
136 }
137
138 // Conexión a mongo
139 var cliente = MongoClientSingleton.Instance.Cliente;
140 if (!await cliente.Conectar())
141 {
142 _dialogoService.MostrarAlerta("Msg_Error_Conexion");
143 EstaCargando = false;
144 return;
145 }
146
147 // Preparamos datos
148 NuevoUsuario.Rol = Roles.Usuario;
149 NuevoUsuario.FechaRegistro = DateTime.Now;
150 NuevoUsuario.Password = Encriptador.HashPassword(NuevoUsuario.Password);
151
152 // Gestionamos foto de perfil
153 try
154 {
155 // Subimos a Imgbb
156 NuevoUsuario.Perfil.ImagenUrl = await _storageService.SubirImagen(NuevoUsuario.Perfil.ImagenUrl);
157 }
158 catch
159 {
160 _dialogoService.MostrarAlerta("Reg_Error_SubirImagen");
161 NuevoUsuario.Perfil.ImagenUrl = "";
162 return;
163 }
164
165 // Si no puso imagen o falló, ponemos una por defecto
166 if (string.IsNullOrEmpty(NuevoUsuario.Perfil.ImagenUrl))
167 {
168 NuevoUsuario.Perfil.ImagenUrl = "https://i.ibb.co/hRJ440cz/image.png";
169 }
170
171 // 6. GUARDAR EN BD
172 bool exito = await cliente.CrearUsuario(NuevoUsuario);
173
174 if (exito)
175 {
176 _dialogoService.MostrarAlerta("Reg_Exito_CuentaCreada");
177 _accionVolver?.Invoke(); // Volvemos al Login automáticamente

183 }
184 catch (Exception ex)

187 }
188 finally
189 {
190 EstaCargando = false;
191 }
