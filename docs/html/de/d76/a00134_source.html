102 private async Task EliminarCancion(Canciones cancion)
103 {
104 // Preguntar antes de borrar para evitar accidentes
105 var confirm = await _dialogoService.Preguntar("MsgConfirmEliminar", "MsgPreguntaEliminarCancion", "BtnEliminar", "BtnCancelar");
106 if (!confirm)
107 {
108 return;
109 }
110 try
111 {
112 // Borrar Audio en la nube (Cloudinary)
113 if (!string.IsNullOrEmpty(cancion.UrlCancion) && cancion.UrlCancion.Contains("cloudinary"))
114 {
115 await _storageService.EliminarArchivo(cancion.UrlCancion);
116 }
117 //Borrar datos en MongoDB
118 bool exito = await MongoClientSingleton.Instance.Cliente.EliminarCancionPorId(cancion.Id);
119
120 if (exito)
121 {
122 //Restar 1 al contador de canciones del usuario en MongoDB
123 await MongoClientSingleton.Instance.Cliente.IncrementarContadorCancionesUsuario(GlobalData.Instance.UserIdGD, -1);
124
125 //Actualizar pantalla
126 MisCanciones.Remove(cancion);
127 _dialogoService.MostrarAlerta("MsgExitoBorrado");

131 _dialogoService.MostrarAlerta("MsgErrorBorradoDB");
