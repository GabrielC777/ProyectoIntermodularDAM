31 /// <summary>
32 /// Solicita de forma asíncrona la extracción y el análisis de metadatos de un recurso de YouTube a través de una API externa.
33 /// </summary>
34 /// <remarks>
35 /// Este método gestiona la comunicación con el microservicio de streaming mediante los siguientes pasos:
36 /// <list type="number">
37 /// <item><b>Construcción:</b> Genera una URI de consulta codificando la <paramref name="urlYoutube"/> como parámetro.</item>
38 /// <item><b>Petición:</b> Realiza una llamada GET utilizando el <c>HttpClient</c> configurado.</item>
39 /// <item><b>Procesamiento:</b> Si la respuesta es exitosa, deserializa el cuerpo JSON para extraer la URL del flujo de audio y la duración total en segundos.</item>
40 /// </list>
41 /// En caso de fallo en la red o error en el servidor, captura la excepción y devuelve <c>null</c> para evitar cierres inesperados.
42 /// </remarks>
43 /// <param name="urlYoutube">La dirección URL completa del video de YouTube que se desea procesar.</param>
44 /// <returns>
45 /// Un objeto <see cref="InfoCancionNube"/> con la URL de streaming y la duración;
46 /// devuelve <c>null</c> si la API no responde correctamente o el recurso no es válido.
47 /// </returns>
48 public async Task<InfoCancionNube> ObtenerMp3(string urlYoutube)
49 {
50 try
51 {
52 // Construimos la petición: https://localhost:7500/api/Music/stream?url=...
53 string urlPeticion = $"{API_URL}?url={urlYoutube}";
54
55 // Llamamos al servidor
56 var respuesta = await _httpClient.GetAsync(urlPeticion);
57
58 if (respuesta.IsSuccessStatusCode)
59 {
60 // Leemos la respuesta (que es un texto JSON)
61 string jsonString = await respuesta.Content.ReadAsStringAsync();
62
63 // Sacamos el valor de "url"
64 var nodoJson = JsonNode.Parse(jsonString);
65
66 return new InfoCancionNube
67 {
68 Url = nodoJson["url"]?.ToString(),
69 // Si 'duracion' viene nulo, ponemos 0
70 DuracionSegundos = nodoJson["duracion"] != null ? (int)nodoJson["duracion"] : 0
71 };
72 }
73 }
74 catch (Exception ex)
75 {
76 System.Diagnostics.Debug.WriteLine("Error llamando a la API: " + ex.Message);
77 }
78
79 return null;

81 /// <summary>
82 /// Gestiona la descarga, protección mediante cifrado y recuperación de archivos de audio en el almacenamiento local.
83 /// </summary>
84 /// <remarks>
85 /// Este método implementa un sistema de seguridad y caché distribuido en dos niveles:
86 /// <list type="number">
87 /// <item><b>Caché persistente (.enc):</b> Si el archivo no existe, se descarga de internet, se cifra mediante AES y se guarda en la carpeta de datos locales de la aplicación.</item>
88 /// <item><b>Caché temporal (.mp3):</b> Para permitir la reproducción en el motor de audio (VLC), el archivo se desencripta "on-the-fly" hacia una ruta temporal volátil.</item>
89 /// <item><b>Optimización:</b> Si el archivo ya ha sido procesado previamente, evita la descarga redundante y prioriza la recuperación desde el disco.</item>
90 /// </list>
91 /// En caso de error crítico durante el cifrado o acceso a disco, el método retorna la URL original como mecanismo de contingencia.
92 /// </remarks>
93 /// <param name="url">La dirección URL de origen del flujo de audio.</param>
94 /// <param name="idCancion">Identificador único de la canción, utilizado para nombrar los archivos en el almacenamiento físico.</param>
95 /// <returns>
96 /// Una cadena con la ruta local al archivo MP3 desencriptado listo para su reproducción,
97 /// o la URL original si ocurre una excepción.
98 /// </returns>
99 public async Task<string> ObtenerRutaAudioSegura(string url, string idCancion)
100 {
101 try
102 {
103 // Definimos rutas
104 // carpetaStorage: Aquí guardaremos los archivos .enc (CIFRADOS). Es persistente.
105 string carpetaStorage = Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData), "BetaProyecto", "EncryptedStorage");
106 // carpetaTemp: Aquí guardaremos temporalmente el .mp3 (DESCIFRADO) para VLC.
107 string carpetaTemp = Path.Combine(Path.GetTempPath(), "BetaProyectoMusicTemp");
108 // Nos aseguramos de que las carpetas existan
109 Directory.CreateDirectory(carpetaStorage);
110 Directory.CreateDirectory(carpetaTemp);
111
112 // Rutas finales de los archivos
113 string rutaCifrada = Path.Combine(carpetaStorage, $"{idCancion}.enc");
114 string rutaPlay = Path.Combine(carpetaTemp, $"{idCancion}.mp3");
115
116 // COMPROBAR CACHÉ (¿Ya la hemos descargado antes?)
117 if (File.Exists(rutaCifrada))
118 {
119 System.Diagnostics.Debug.WriteLine($"[CACHE] Canción cifrada encontrada en disco.");
120
121 // Si ya tenemos el temporal listo, lo usamos directo para ir rápido
122 if (File.Exists(rutaPlay)) return rutaPlay;
123
124 // Si no está el temporal, usamos el Encriptador para "abrir el candado"
125 // Lee el .enc -> Usa la clave AES -> Escribe el .mp3
126 await Helpers.Encriptador.DesencriptarArchivo(rutaCifrada, rutaPlay);
127
128 return rutaPlay;
129 }
130
131 // DESCARGAR Y PROTEGER (Si no la tenemos)
132 System.Diagnostics.Debug.WriteLine($"[RED] Descargando audio...");
133
134 // Descargamos los bytes crudos de internet
135 var datosAudio = await _httpClient.GetByteArrayAsync(url);
136
137 System.Diagnostics.Debug.WriteLine($"[SEGURIDAD] Cifrando con AES...");
138
139 // Usamos el Encriptador para "cerrar el candado" antes de guardar
140 byte[] datosCifrados = Helpers.Encriptador.EncriptarBytes(datosAudio);
141
142 // Guardamos la versión CIFRADA (Esta es la que se queda en el disco duro)
143 await File.WriteAllBytesAsync(rutaCifrada, datosCifrados);
144
145 // Guardamos la versión DESCIFRADA temporalmente (Solo para que VLC la reproduzca ahora)
146 await File.WriteAllBytesAsync(rutaPlay, datosAudio);
147
148 return rutaPlay;
149 }
150 catch (Exception ex)
151 {
152 System.Diagnostics.Debug.WriteLine($"[ERROR] Fallo en descarga segura: {ex.Message}");
153 // Si algo falla, devolvemos la URL original para intentar streaming directo que lo mas seguro es que falle,
154 // pero al menos no se crasea la app.
155 return url;
156 }
157 }
